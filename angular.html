angular.bind(self, fn, args); 数据绑定 ，作用同 --> {{}}
angular.bootstrap(element, [modules], [config]); 手动启动angular
angular.copy(source, [destination]); 数组或对象深度拷贝
angular.element(element); 包裹着一部分DOM element或者是HTML字符串，把它作为一个jQuery元素来处理。
angular.equals(o1, o2); 如果两个对象相等返回 true
angular.extend(dst, src); 复制src对象中的属性去dst对象中. 支持多个src对象. 如果你不想改变一个对象，你可以把dst设为空对象{}: var object = angular.extend({}, object1, object2). 注意: angular.extend不支持递归复制.
angular.forEach(obj, iterator, [context]); 循环对obj对象的每个元素调用iterator, obj对象可以是一个Object或一个Array. Iterator函数调用方法: iterator(value, key, obj), 其中obj是被迭代对象，key是obj的property key或者是数组的index，value就是相应的值啦. (此函数不能够迭代继承的属性.)
angular.fromJson(json); 反序列化JSON字符串。
angular.identity(value); 函数返回本身的第一个参数。这个函数一般用于函数风格。
angular.injector(modules, [strictDi]);
angular.isArray(value); 如果引用的是数组返回 true
angular.isDate(value); 如果引用的是日期返回 true
angular.isDefined(value); 如果引用的已定义返回 true
angular.isElement(value);
angular.isFunction(value);
angular.isNumber(value);
angular.isObject(value);
angular.isString(value);
angular.isUndefined(value);
angular.lowercase(string);
angular.module(name, [requires], [configFn]);
angular.noop(); 一个不执行任何操作的空函数。这个函数一般用于函数风格。
